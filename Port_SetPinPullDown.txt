#include "typedefs.h"
#include "dio.h"
#include "port.h"
#include "tm4c123gh6pm.h"
void Port_SetPinPullDown( uint8 port_index ,uint8 pins_mask ,uint8 enable)
{
if ( enable ==1 )
{
	if(port_index == PORTA_INDEX)		GPIOA->PDR &= pins_mask; 
	else if(port_index == PORTB_INDEX)	GPIOB->PDR &= pins_mask; 
	else if(port_index == PORTC_INDEX)	GPIOC->PDR &= pins_mask;  
	else if(port_index == PORTD_INDEX)	GPIOD->PDR &= pins_mask; 
	else if(port_index == PORTE_INDEX)	GPIOE->PDR &= pins_mask; 
	else if(port_index == PORTF_INDEX)	GPIOF->PDR &= pins_mask; 
  }
else 
{
	
	if(port_index == PORTA_INDEX)		GPIOA->PDR &= (~pins_mask); 
	else if(port_index == PORTB_INDEX)	GPIOB->PDR &= (~pins_mask); 
	else if(port_index == PORTC_INDEX)	GPIOC->PDR &= (~pins_mask);  
	else if(port_index == PORTD_INDEX)	GPIOD->PDR &= (~pins_mask); 
	else if(port_index == PORTE_INDEX)	GPIOE->PDR &= (~pins_mask); 
	else if(port_index == PORTF_INDEX)	GPIOF-> PDR &=(~pins_mask);

}
